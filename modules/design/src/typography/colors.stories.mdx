export const ColorBox = ({ color }) => {
  const { theme } = useUIConfig()
  const rgba = theme.colors[color]
  const [r,g,b,a] = parseRGBA(rgba)
  const hex = `#${r.toString(16)}${g.toString(16)}${b.toString(16)}`
  return (
    <Flex
      column
      justify="center"
      align="center"
      width={160}
      height={80}
      padding={5}
      shrink={0}
      style={{ backgroundColor: theme.colors[color], borderRadius: 8 }}
    >
      <Text.Bold>{color}</Text.Bold>
      <Text.Small style={{ marginTop: 5 }}>{hex}</Text.Small>
      <Text.Small style={{ marginTop: 5 }}>{a}</Text.Small>
    </Flex>
  )
}
